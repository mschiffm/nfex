#import <Foundation/Foundation.h>
#import <pcap.h>
#import <net/ethernet.h>
#import <netinet/ip.h>
#import <netinet/ip_icmp.h>
#import <netinet/tcp.h>
#import <netinet/udp.h>

@interface Packet : NSObject
{
    pcap_t *p;
    unsigned char *packet;
}

@property pcap_t *p;
@property unsigned char *packet;

-(int)  initializePcap:(char *)device
            withFilter:(char *)packetFilter;

-(void) capturePackets;

-(void) processIPV4Packet:(unsigned char *)IPPacket;
-(void) processIPV6Packet:(unsigned char *)IPPacket;
-(void) processARPPacket:(unsigned char *)ARPPacket;
-(void) processTCPPacket:(unsigned char *)TCPPacket;
-(void) processUDPPacket:(unsigned char *)UDPPacket;
-(void) processICMPPacket:(unsigned char *)ICMPPacket;

@end
